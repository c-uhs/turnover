% parts
\newcommand{\drawnodes}{
  % main 3 nodes
  \node(x1) [tbox=\colorx] at (0.0,0.0)         {\small{High}};
  \node(x2) [tbox=\colorx, below = 1.5cm of x1] {\small{Med}};
  \node(x3) [tbox=\colorx, below = 1.5cm of x2] {\small{Low}};
  % input nodes (no tbox)
  \node(i1) [left = 2cm of x1] {};
  \node(i2) [left = 2cm of x2] {};
  \node(i3) [left = 2cm of x3] {};
  % output nodes (no tbox)
  \node(o1) [right = 2cm of x1] {};
  \node(o2) [right = 2cm of x2] {};
  \node(o3) [right = 2cm of x3] {};
}
\newcommand{\drawturnover}{
  \begin{pgfonlayer}{background}
  \draw[arrow=\colort] (x1) edge[bend left=25] (x2); 
  \draw[arrow=\colort] (x2) edge[bend left=25] (x1);
  \draw[arrow=\colort] (x3) edge[bend left=25] (x2); 
  \draw[arrow=\colort] (x2) edge[bend left=25] (x3);
  \draw[arrow=\colort] (x1) edge[bend left=60, near end] (x3); 
  \draw[arrow=\colort] (x3) edge[bend left=60, near end] (x1);
  \end{pgfonlayer}
  \node(l3) [legend, right = 0.2 of l2] {\footnotesize Turnover};
  \draw[arrow=\colort,-] (l3) -- ++ (-1.3cm,0);
}

}
\newcommand{\drawexit}{
